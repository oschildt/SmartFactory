107:     /**
108:      * Formats the standard PHP backtrace (debug_backtrace).
109:      *
110:      * @param array $btrace
111:      * The backtrace.
112:      *
113:      * @return string
114:      * Returns the formatted backtrace.
115:      *
116:      * @author Oleg Schildt
117:      */
118:     protected function format_backtrace($btrace)
119:     {
120:         if (empty($btrace) || count($btrace) == 0) {
121:             return "backtrace empty";
122:         }
123:         
124:         $trace = "";
125:         
126:         foreach ($btrace as $nr => &$btrace_entry) {
127:             if ($nr == 0) {
128:                 $trace .= $btrace_entry["args"][2] . "\r\n" . "line: " . $btrace_entry["args"][3] . "\r\n" . $btrace_entry["args"][1];
129:                 
130:                 $cstack = $this->extract_call_stack($btrace);
131:                 if (!empty($cstack)) {
132:                     $trace .= "\r\n\r\nCall stack:\r\n\r\n" . $cstack;
133:                 }
134:                 
135:                 continue;
136:             }
137:         }
138:         
139:         return $trace;
140:     } // format_backtrace

142:     /**
143:      * Extracts the call stack from the standard PHP backtrace (debug_backtrace).
144:      *
145:      * @param array $btrace
146:      * The backtrace.
147:      *
148:      * @return string
149:      * Returns the extracted call stack.
150:      *
151:      * @author Oleg Schildt
152:      */
153:     protected function extract_call_stack($btrace)
154:     {
155:         if (empty($btrace) || !is_array($btrace)) {
156:             return "";
157:         }
158:         
159:         $trace = "";
160:         
161:         $indent = "";
162:         foreach ($btrace as $btrace_entry) {
163:             
164:             
165:             if (!empty($btrace_entry["function"]) && ($btrace_entry["function"] == "handle_error" || strpos($btrace_entry["function"], "{closure}") !== false || $btrace_entry["function"] == "handleError" || $btrace_entry["function"] == "trigger_error")) {
166:                 continue;
167:             }
168:             
169:             if (empty($btrace_entry["file"])) {
170:                 continue;
171:             }
172:             
173:             if (!empty($btrace_entry["function"])) {
174:                 $trace .= $indent . str_replace("SmartFactory\\", "", $btrace_entry["function"]) . "() ";
175:             }
176:             
177:             $trace .= "[";
178:             
179:             $trace .= str_replace($this->app_root, "", str_replace("\\", "/", $btrace_entry["file"]));
180:             
181:             $trace .= ", ";
182:             
183:             if (empty($btrace_entry["line"])) {
184:                 $trace .= "line number undefined";
185:             } else {
186:                 $trace .= $btrace_entry["line"];
187:             }
188:             
189:             $trace .= "]";
190:             
191:             $args = (isset($btrace_entry["args"])) ? $btrace_entry["args"] : [];
192:             $args_str = $this->make_arg_list($args);
193:             
194:             if (!empty($btrace_entry["function"]) && !empty($args_str)) {
195:                 $trace .= "  " . str_replace("SmartFactory\\", "", $btrace_entry["function"]) . "(" . $args_str . ")";
196:             }
197:             
198:             $trace .= "\r\n";
199:             
200:             $indent .= "  ";
201:         }
202:         
203:         return trim($trace);
204:     } // extract_call_stack

282:     protected function trace_message($message)
283:     {
284:         if (!$this->traceActive()) {
285:             return true;
286:         }
287:

289:         
290:         if ((!file_exists($file) && is_writable($this->log_path)) || is_writable($file)) {
291:             return file_put_contents($file, date("Y-m-d H:i:s") . "\r\n" .
292:                 "----------------------------------------------------------\r\n" .
293:                 $message . "\r\n" .
294:                 "----------------------------------------------------------\r\n" .
295:                 "\r\n\r\n", FILE_APPEND) !== false;
296:         } else {
297:             throw new \Exception(sprintf("The trace file '%s' is not writable!", $file));

301:     /**
302:      * This is the function for handling of the PHP errors. It is set a the
303:      * error handler.
304:      *
305:      * @param int $errno
306:      * Error code.
307:      *
308:      * @param string $errstr
309:      * Error text.
310:      *
311:      * @param string $errfile
312:      * Source file where the error occured.
313:      *
314:      * @param int $errline

320:      */
321:     public function handleError($errno, $errstr, $errfile, $errline)
322:     {
323:         $this->setLastError($errstr);
324:         
325:         $errortype = [
326:             E_ERROR => "Error",
327:             E_WARNING => "Warning",
328:             E_PARSE => "Parsing Error",
329:             E_NOTICE => "Notice",
330:             E_CORE_ERROR => "Core Error",
331:             E_CORE_WARNING => "Core Warning",
332:             E_COMPILE_ERROR => "Compile Error",
333:             E_COMPILE_WARNING => "Compile Warning",
334:             E_USER_ERROR => "User Error",
335:             E_USER_WARNING => "User Warning",
336:             E_USER_NOTICE => "User Notice",
337:             E_STRICT => "Runtime Notice",
338:             E_DEPRECATED => "Deprecated Notice"
339:         ];
340:         
341:         if (empty($errortype[$errno])) {
342:             $etype = $errno;
343:         } else {
344:             $etype = $errortype[$errno];
345:         }
346:         
347:         $this->trace_message($this->format_backtrace(debug_backtrace()));
348:         
349:         event()->fireEvent("php_error", ["etype" => $etype, "errstr" => $errstr, "errfile" => $errfile, "errline" => $errline]);
350:     } // handleError

353:      * This is the function for handling of the PHP exceptions. It should

373:     public function handleException($ex, $errfuntion, $errfile, $errline)
374:     {
375:         $this->setLastError($ex->getMessage());
376:         
377:         $errortype = [
378:             E_ERROR => "Error",
379:             E_WARNING => "Warning",
380:             E_PARSE => "Parsing Error",
381:             E_NOTICE => "Notice",
382:             E_CORE_ERROR => "Core Error",
383:             E_CORE_WARNING => "Core Warning",
384:             E_COMPILE_ERROR => "Compile Error",
385:             E_COMPILE_WARNING => "Compile Warning",
386:             E_USER_ERROR => "User Error",
387:             E_USER_WARNING => "User Warning",
388:             E_USER_NOTICE => "User Notice",
389:             E_STRICT => "Runtime Notice",
390:             E_DEPRECATED => "Deprecated Notice"
391:         ];
392:         
393:         $errno = $ex->getCode();
394:         
395:         if (empty($errortype[$errno])) {
396:             $etype = $errno;
397:         } else {
398:             $etype = $errortype[$errno];
399:         }
400:         
401:         $trace = $ex->getTrace();
402:         $trace_entry["args"] = ["", $ex->getMessage(), $errfile, $errline];
403:         array_unshift($trace, $trace_entry);
404:         
405:         $this->trace_message($this->format_backtrace($trace));
406:         
407:         event()->fireEvent("php_error", ["etype" => $etype, "errstr" => $ex->getMessage(), "errfile" => $ex->getFile(), "errline" => $ex->getLine()]);
408:     }
