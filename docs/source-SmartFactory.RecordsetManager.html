181:     protected function processSubarray(&$subarray, $key_fields, &$parent_values, &$record)
182:     {
183:         $current_key = array_shift($key_fields);
184:         
185:         foreach ($subarray as $key => $value) {
186:             if (!empty($current_key)) {
187:                 $record[$current_key] = $key;
188:                 
189:                 if (!empty($parent_values[$current_key])) {
190:                     $record[$current_key] = $parent_values[$current_key];
191:                 }
192:                 
193:                 if (!$this->processSubarray($value, $key_fields, $parent_values, $record)) {
194:                     return false;
195:                 }
196:             } else {
197:                 $record[$key] = $value;
198:             }
199:         }
200:         
201:         if (empty($current_key)) {
202:             return $this->saveRecord($record);
203:         }

416:         $query .= implode(", ", array_keys($this->fields)) . "\n";
417:         
418:         $query .= "FROM " . $this->table . "\n";
419:         
420:         if (!empty($where_clause)) {
421:             $query .= $where_clause . "\n";
422:         }
423:

429:

445:             
446:             foreach ($dimensions as &$dval) {
447:                 if (empty($reference[$dval])) {

451:             }

463:     /**
464:      * Saves a record from an array in the form "field_name" => "value" into the table.
465:      *
466:      * @param array $record
467:      * The source array with the data to be saved.
468:      *
469:      * @param string $identity_field
470:      * The name of the identity field if exists. If the identity field is specified
471:      * and the record does not exist yet in the table, the source array is extended
472:      * with a pair "identity field" => "identity value" issued by the database by this
473:      * insert operation.
474:      *
475:      * @return boolean
476:      * Returns true if the record has been successfully saved, otherwise false.
477:      *
478:      * @throws \Exception
479:      * It might throw an exception in the case of any errors:
480:      *
481:      * - if some parameters are missing.
482:      * - if dbworker does not extend {@see \SmartFactory\DatabaseWorkers\DBWorker}.
483:      * - if some parameters are not of the proper type.
484:      * - if the query fails or if some object names are invalid.
485:      *
486:      * @see  loadRecord()
487:      * @see  saveRecordSet()
488:      *
489:      * @uses DatabaseWorkers\DBWorker
490:      *
491:      * @author Oleg Schildt
492:      */
493:     public function saveRecord(&$record, $identity_field = null)
494:     {
495:         $this->validateParameters();
496:         
497:         $where = "";
498:         $must_insert = false;
499:         
500:         // key_fields not specified - always insert
501:         // identity firld exists but its value is not specified - always insert
502:         if (empty($this->key_fields) || (!empty($identity_field) && empty($record[$identity_field]))) {
503:             $must_insert = true;
504:         } else {
505:             // check existence
506:             
507:             $query = "SELECT\n";
508:             
509:             $query .= "1\n";
510:             
511:             $query .= "FROM " . $this->table . "\n";
512:             
513:             foreach ($this->key_fields as $key_field) {
514:                 if (!empty($where)) {
515:                     $where .= " AND ";
516:                 }
517:                 
518:                 $value = $this->dbworker->prepare_for_query(checkempty($record[$key_field]), checkempty($this->fields[$key_field]));
519:                 
520:                 if ($value == "NULL") {
521:                     $where .= $key_field . " IS NULL";
522:                 } else {
523:                     $where .= $key_field . " = " . $value;
524:                 }
525:             }
526:             
527:             if (!empty($where)) {
528:                 $where = "WHERE " . $where;
529:             }
530:             
531:             $query .= $where;
532:             
533:             $this->dbworker->execute_query($query);
534:             
535:             if (!$this->dbworker->fetch_row()) {
536:                 $must_insert = true;
537:             }
538:             
539:             $this->dbworker->free_result();
540:         }
541:         
542:         // saving
543:         
544:         $update_string = "";
545:         $insert_fields = "";
546:         $insert_values = "";
547:         
548:         foreach ($this->fields as $field => $type) {
549:             if ($must_insert && $field == $identity_field) {
550:                 continue;
551:             }
552:             
553:             if (!$must_insert && in_array($field, $this->key_fields)) {
554:                 continue;
555:             }
556:             
557:             // The value for the field is not passed in the record,
558:             // skip it
559:             if (!array_key_exists($field, $record)) {
560:                 continue;
561:             }
562:             
563:             $value = $this->dbworker->prepare_for_query(checkempty($record[$field]), checkempty($this->fields[$field]));
564:             
565:             $update_string .= $field . " = " . $value . ",\n";
566:             $insert_fields .= $field . ", ";
567:             $insert_values .= $value . ", ";
568:         }
569:         
570:         if ($must_insert) {
571:             $query = "INSERT INTO " . $this->table . "(" . trim($insert_fields, ", ") . ")\n";
572:             $query .= "VALUES (" . trim($insert_values, ", ") . ")\n";
573: 
574:             $this->dbworker->execute_query($query);
575:         } elseif(!empty($update_string)) {
576:             $query = "UPDATE " . $this->table . " SET\n";
577:             $query .= trim($update_string, ",\n") . "\n";
578:             $query .= $where;
579: 
580:             $this->dbworker->execute_query($query);
581:         }
582:         
583:         if (!empty($identity_field) && $must_insert) {
584:             $record[$identity_field] = $this->dbworker->insert_id();
585:         }
586:         
587:         return true;
588:     } // saveRecord
