201:     /**
202:      * Starts the session.
203:      *
204:      * @param boolean $readonly
205:      * This paramters specifies whether the session should be started
206:      * in the readonly mode or not.
207:      *
208:      * The parameter $readonly starts the session in non-blocking
209:      * readonly mode. It can be used in the asynchronous ajax requests
210:      * So that they are not blocked by the main process and by each
211:      * ohter while the write lock is held on the session file.
212:      *
213:      * @param string $context
214:      * The session context.
215:      *
216:      * If many instances of the application should run in parallel
217:      * subfolders, and all subfolders are within the same session,
218:      * and the provider does not let you to change the session path,
219:      * then you can use different $context in each instance to ensure
220:      * that the session data of these instances does not mix.
221:      *
222:      * @return boolean
223:      * Returns true if the session has been successfully started, otherwise false.
224:      *
225:      * @author Oleg Schildt
226:      */
227:     public function startSession($readonly = false, $context = "default")
228:     {
229:         self::$context = $context;
230:         self::$readonly = $readonly;
231:         
232:         if (!$readonly) {
233:             return session_start();
234:         }
235:         
236:         // we read directly from the session file without blocking it
237:         
238:         global $_SESSION;
239:         
240:         $session_path = session_save_path();
241:         if (empty($session_path)) {
242:             $session_path = sys_get_temp_dir();
243:         }
244:         
245:         $session_path = rtrim($session_path, '/\\');
246:         
247:         if (empty($_COOKIE[session_name()])) {
248:             session_start();
249:             session_write_close();
250:             return false;
251:         }
252:         
253:         $session_name = preg_replace('/[^\da-z]/i', '', $_COOKIE[session_name()]);
254:         
255:         if (!file_exists($session_path . '/sess_' . $session_name)) {
256:             session_start();
257:             session_write_close();
258:             return false;
259:         }
260:         
261:         $session_data = file_get_contents($session_path . '/sess_' . $session_name);
262:         if (empty($session_data)) {
263:             session_start();
264:             session_write_close();
265:             return false;
266:         }
267:         
268:         $offset = 0;
269:         
270:         while ($offset < strlen($session_data)) {
271:             if (!strstr(substr($session_data, $offset), "|")) {
272:                 break;
273:             }
274:             
275:             $pos = strpos($session_data, "|", $offset);
276:             $num = $pos - $offset;
277:             $varname = substr($session_data, $offset, $num);
278:             $offset += $num + 1;
279:             $data = unserialize(substr($session_data, $offset));
280:             $_SESSION[$varname] = $data;
281:             $offset += strlen(serialize($data));
282:         }
283:         
284:         return true;
285:     } // startSession

456:     /**
457:      * Returns the reference to the array of the session variables.
458:      *
459:      * @return array
460:      * Returns the reference to the array of the session variables.
461:      *
462:      * The reason of this interacfe is to wrap the standard session
463:      * handling to be able to flexibly change the implementation without
464:      * any change in the code where it is used.
465:      *
466:      * Thus, we cannot use $_SESSION directly, a new implementation
467:      * might not use it at all. But we need a comfort way to set and
468:      * get the session variables including the multidimesional arrays.
469:      *
470:      * Operator overloading through ArrayAccess interface supports
471:      * only one dimensional arrays, thus, it is not applicable for us.
472:      *
473:      * We return the reference to the internal data. It allows setting
474:      * multidimesional arrays as follows:
475:      *
476:      * ```php
477:      * $smanager->vars()["user"]["name"] = "Alex";
478:      * $smanager->vars()["user"]["age"] = "22";
479:      * ```
480:      *
481:      * @author Oleg Schildt
482:      */
483:     public function &vars()
484:     {
485:         if (empty($_SESSION[$this->getContext()])) {
486:             $_SESSION[$this->getContext()] = [];
487:         }
488:         
489:         return $_SESSION[$this->getContext()];
