 78:     /**
 79:      * This is an auxiliary function for parsing the incoming XML data.
 80:      *
 81:      * It validates the content type 'application/xml' and takes the data from RAWDATA.
 82:      *
 83:      * @throws SmartException
 84:      * It might throw an exception if the content type o–∫ XML data is invalid.
 85:      *
 86:      * @return void
 87:      *
 88:      * @author Oleg Schildt
 89:      */
 90:     protected function parseInput()
 91:     {
 92:         try {
 93:             $content_type = get_header("Content-Type");
 94: 
 95:             if (empty($content_type)) {
 96:                 throw new SmartException("Content type header 'Content-Type' is missing, expected 'application/xml; charset=UTF-8'!", SmartException::ERR_CODE_INVALID_CONTENT_TYPE);
 97:             }
 98: 
 99:             if (!preg_match("/application\/xml.*/", $content_type)) {
100:                 throw new SmartException(sprintf("Content type 'application/xml; charset=UTF-8' is expected, got '%s'!", $content_type), SmartException::ERR_CODE_INVALID_CONTENT_TYPE);
101:             }
102: 
103:             $xmldata = trim(file_get_contents("php://input"));
104: 
105:             if (empty($xmldata)) {
106:                 throw new SmartException("The request XML is empty!", SmartException::ERR_CODE_MISSING_REQUEST_DATA);
107:             }
108: 
109:             $this->request_xmldoc = new \DOMDocument("1.0", "UTF-8");
110:             $this->request_xmldoc->formatOutput = true;
111:             if (!@$this->request_xmldoc->loadXML($xmldata)) {
112:                 throw new SmartException("Error by XML parsing!", SmartException::ERR_CODE_XML_PARSE_ERROR);
113:             }
114:         } catch (SmartException $ex) {
115:             throw $ex;
116:         } catch (\Throwable $ex) {
117:             throw new SmartException($ex->getMessage(), SmartException::ERR_CODE_XML_PARSE_ERROR);
118:         }
119:     } // parseInput

121:     /**
122:      * This is an auxiliary function for sending the response in XML format.
123:      *
124:      * It sends the prepared response XML document and the response headers. The header
125:      * "Content-type: application/xml" ist sent automatically.
126:      *
127:      * @return void
128:      *
129:      * @author Oleg Schildt
130:      */
131:     protected function sendResponse()
132:     {
133:         $this->addResponseHeader("Content-type", "application/xml; charset=UTF-8");
134: 
135:         header('Content-type: application/xml');
136: 
137:         if (!empty($this->response_headers)) {
138:             if (is_array($this->response_headers)) {
139:                 foreach ($this->response_headers as $header) {
140:                     header($header);
141:                 }
142:             }
143:         }
144: 
145:         $this->addMessagesToResponse();

147:         echo $this->response_xmldoc->saveXML();
148:     } // sendResponse
