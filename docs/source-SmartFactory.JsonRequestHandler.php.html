 14: /**
 15:  * Class for handling the JSON API requests.
 16:  *
 17:  * The user should derive from this class and implement the methods with the names
 18:  * of the API actions.
 19:  *
 20:  * URL: http://smartfactorydev.loc/api/login/
 21:  *
 22:  * Action: login
 23:  *
 24:  * ```php
 25:  * function login()
 26:  * {
 27:  *     if (empty($this->request_data["login"]) || empty($this->request_data["password"]) ||
 28:  *         $this->request_data["login"] != "admin" || $this->request_data["password"] != "qwerty") {
 29:  *         throw new SmartException("Wrong login or password!", "wrong_login");
 30:  *     }
 31:  *
 32:  *     $this->response_data["welcome_msg"] = "Welcome, admin!";
 33:  * }
 34:  * ```
 35:  *
 36:  * @author Oleg Schildt
 37:  */
 38: abstract class JsonRequestHandler extends RequestHandler
 39: {
 40:     /**
 41:      * Internal variable for storing the request data.
 42:      *
 43:      * @var array
 44:      *
 45:      * @author Oleg Schildt
 46:      */
 47:     protected $request_data = [];
 48: 
 49:     /**
 50:      * Internal variable for storing the response data.
 51:      *
 52:      * @var array
 53:      *
 54:      * @author Oleg Schildt
 55:      */
 56:     protected $response_data = [];
 57: 
 58:     /**
 59:      * This is an auxiliary function for sending the response in JSON format.
 60:      *
 61:      * It sends the collected response data and the response headers. The header
 62:      * "Content-type: application/json; charset=UTF-8" ist sent automatically.
 63:      *
 64:      * @return void
 65:      *
 66:      * @author Oleg Schildt
 67:      */
 68:     protected function sendResponse()
 69:     {
 70:         $this->addResponseHeader("Content-type", "application/json; charset=UTF-8");
 71: 
 72:         if (!empty($this->response_headers)) {
 73:             if (is_array($this->response_headers)) {
 74:                 foreach ($this->response_headers as $header => $value) {
 75:                     header($header . ": " . $value);
 76:                 }
 77:             }
 78:         }
 79: 
 80:         $this->addMessagesToResponse();
 81: 
 82:         if (empty($this->response_data)) {
 83:             echo json_encode($this->response_data, JSON_FORCE_OBJECT | JSON_PRETTY_PRINT);
 84:         } else {
 85:             echo json_encode($this->response_data, JSON_PRETTY_PRINT | JSON_UNESCAPED_UNICODE | JSON_UNESCAPED_SLASHES);
 86:         }
 87:     } // sendResponse
 88: 
 89:     /**
 90:      * This is an auxiliary function for parsing the incoming JSON data.
 91:      *
 92:      * It validates the content type 'application/json' and takes the data from RAWDATA.
 93:      *
 94:      * @return void
 95:      *
 96:      * @throws SmartException
 97:      * It might throw an exception if the content type o–∫ JSON data is invalid.
 98:      *
 99:      * @author Oleg Schildt
100:      */
101:     protected function parseInput()
102:     {
103:         try {
104:             $content_type = get_header("Content-Type");
105:             
106:             if (empty($content_type)) {
107:                 throw new SmartException("Content type header 'Content-Type' is missing, expected 'application/json; charset=UTF-8'!", SmartException::ERR_CODE_INVALID_CONTENT_TYPE);
108:             }
109: 
110:             if (!preg_match("/application\/json.*charset=UTF-8/i", $content_type)) {
111:                 throw new SmartException(sprintf("Content type 'application/json; charset=UTF-8' is expected, got '%s'!", $content_type), SmartException::ERR_CODE_INVALID_CONTENT_TYPE);
112:             }
113: 
114:             $json = trim(file_get_contents("php://input"));
115: 
116:             if (empty($json)) {
117:                 throw new SmartException("The request JSON is empty!", SmartException::ERR_CODE_MISSING_REQUEST_DATA);
118:             }
119: 
120:             $json = json_decode($json, true);
121: 
122:             if (empty($json) && !is_array($json)) {
123:                 throw new SmartException(json_last_error_msg(), SmartException::ERR_CODE_JSON_PARSE_ERROR);
124:             }
125: 
126:             $this->request_data = array_merge($this->request_data, $json);
127:         } catch (SmartException $ex) {
128:             throw $ex;
129:         } catch (\Throwable $ex) {
130:             throw new SmartException($ex->getMessage(), SmartException::ERR_CODE_JSON_PARSE_ERROR);
131:         }
132:     } // parseInput
133: } // JsonRequestHandler
