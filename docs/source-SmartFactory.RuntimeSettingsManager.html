 17: /**
 18:  * Class for management of the application settings.
 19:  *
 20:  * This settings manager supports the multistep wizard like processing. That means
 21:  * that you do not have to put all settings in one big form and save them in one action.
 22:  * You can collect them over multiple steps (requests), go forward ad back, validate them
 23:  * on each step and finally save them after the final step.
 24:  *
 25:  * You do not need any preliminary special settings name definitions. When you introduce
 26:  * a new setting, just start saving and getting it. But you need provide a settings table
 27:  * with a column where the settings data will be saved.
 28:  *
 29:  * @see  ConfigSettingsManager
 30:  * @see  UserSettingsManager

137:     /**
138:      * This is internal auxiliary function for converting the settings to JSON and storing it
139:      * to the target table defined by the iniailization.
140:      *
141:      * @param array $data
142:      * The array with the settings values to be saved.
143:      *
144:      * @return boolean
145:      * Returns true if the data has been successfully saved, otherwise false.
146:      *
147:      * @throws \Exception
148:      * It might throw an exception in the case of any errors:
149:      *
150:      * - if some parameters are missing.
151:      * - if dbworker does not extend {@see \SmartFactory\DatabaseWorkers\DBWorker}.
152:      * - if the query fails or if some object names are invalid.
153:      *
154:      * @see loadJSON()
155:      *
156:      * @author Oleg Schildt
157:      */
158:     protected function saveJSON(&$data)
159:     {
160:         $this->validateParameters();
161:         
162:         $json = array_to_json($data);
163:         
164:         // check existance
165:         
166:         $query = "SELECT 1 FROM " . $this->settings_table;
167:         
168:         $this->dbworker->execute_query($query);
169:         
170:         $must_insert = true;
171:         
172:         if ($this->dbworker->fetch_row()) {
173:             $must_insert = false;
174:         }
175:         
176:         $this->dbworker->free_result();
177:         
178:         $json = $this->dbworker->escape($json);
179:         
180:         if ($must_insert) {
181:             $query = "INSERT INTO " . $this->settings_table . "(" . $this->settings_column . ")\n";
182:             $query .= "VALUES ('$json')";
183:         } else {
184:             $query = "UPDATE " . $this->settings_table . " SET " . $this->settings_column . " = '$json'";
185:         }
186:         
187:         $this->dbworker->execute_query($query);
188:         
189:         return true;
190:     } // saveJSON

193:      * This is internal auxiliary function for loading the settings from the target table
194:      * defined by the iniailization.
195:      *
196:      * @param array $data

199:      * @return boolean

203:      * It might throw an exception in the case of any errors:
204:      *
205:      * - if some parameters are missing.
206:      * - if dbworker does not extend {@see \SmartFactory\DatabaseWorkers\DBWorker}.
207:      * - if the query fails or if some object names are invalid.
208:      * - if the json is invalid.
209:      *
210:      * @see saveJSON()
211:      *
212:      * @author Oleg Schildt
213:      */
214:     protected function loadJSON(&$data)
215:     {
216:         $this->validateParameters();
217:         
218:         $query = "SELECT " . $this->settings_column . " FROM " . $this->settings_table;
219:         
220:         $this->dbworker->execute_query($query);
221:         
222:         $json = "";
223:         
224:         if ($this->dbworker->fetch_row()) {
225:             $json = $this->dbworker->field_by_name($this->settings_column);
226:         }
227:         
228:         $this->dbworker->free_result();
229:         
230:         if (empty($json)) {
231:             return true;
232:         }
233:         
234:         try {
235:             json_to_array($json, $data);
236:         } catch (\Exception $ex) {
237:             throw new \Exception("JSON parse error: " . $ex->getMessage());
238:         }
239:         
240:         return true;
241:     } // loadJSON
