105:      * This is an auxiliary function for getting request headers and parsing the API url.
106:      *
107:      * It put the retrieved data into the corresponding properties.
108:      *
109:      * @return void
110:      *

117:     protected function processInputData()
118:     {
119:         if (empty($_SERVER['REQUEST_URI'])) {
120:             return;
121:         }
122: 
123:         $api_base = str_replace(basename($_SERVER['SCRIPT_NAME']), "", $_SERVER['SCRIPT_NAME']);
124:         $api_request = str_replace($api_base, "", $_SERVER['REQUEST_URI']);
125: 
126:         if (preg_match("~^([^/.?&]+)([/?]?.+)~", $api_request, $matches)) {
127:             $this->action = $matches[1];
128:             $this->param_string = $matches[2];
129:         } else {
130:             $this->action = reqvar_value("action");
131:             $this->param_string = str_replace(basename($_SERVER['SCRIPT_NAME']), "", $api_request);
132:         }
133: 
134:         $this->request_headers = getallheaders();
135:     } // processInputData

137:     /**
138:      * This is an auxiliary function for sending the response in JSON format.
139:      *

149:     protected function sendJsonResponse()
150:     {
151:         header('Content-type: application/json');
152: 
153:         if (!empty($this->response_headers)) {
154:             if (is_array($this->response_headers)) {
155:                 foreach ($this->response_headers as $header) {
156:                     header($header);
157:                 }
158:             }
159:         }

161:         if (empty($this->response_data)) {
162:             echo json_encode($this->response_data, JSON_FORCE_OBJECT | JSON_PRETTY_PRINT);
163:         } else {
164:             echo json_encode($this->response_data, JSON_PRETTY_PRINT | JSON_NUMERIC_CHECK | JSON_UNESCAPED_UNICODE);

252:     /**
253:      * This is an auxiliary function for parsing the incoming JSON data.
254:      *
255:      * It validates the content type 'application/json' and takes the data from RAWDATA.
256:      *
257:      * @throws SmartException
258:      * It might throw an exception if the content type o–∫ JSON data is invalid.
259:      *
260:      * @return void
261:      *
262:      * @author Oleg Schildt
263:      */
264:     protected function parseJsonInput()
265:     {
266:         try {
267:             $content_type = empty($this->request_headers["Content-Type"]) ? "" : $this->request_headers["Content-Type"];
268:             if (!preg_match("/application\/json.*/", $content_type)) {
269:                 throw new SmartException(sprintf("Content type 'application/json' is expected, got '%s'!", $content_type), SmartException::ERR_CODE_INVALID_CONTENT_TYPE, SmartException::ERR_TYPE_PROGRAMMING_ERROR);
270:             }
271: 
272:             $json = trim(file_get_contents("php://input"));
273: 
274:             if (empty($json)) {
275:                 throw new SmartException("The request JSON is empty!", SmartException::ERR_CODE_MISSING_REQUEST_DATA, SmartException::ERR_TYPE_PROGRAMMING_ERROR);
276:             }
277: 
278:             $json = json_decode($json, true);
279: 
280:             if (empty($json) && !is_array($json)) {
281:                 throw new SmartException(json_last_error_msg(), SmartException::ERR_CODE_JSON_PARSE_ERROR, SmartException::ERR_TYPE_PROGRAMMING_ERROR);
282:             }
283: 
284:             $this->request_data = array_merge($this->request_data, $json);
285:         } catch (SmartException $ex) {
286:             throw $ex;
287:         } catch (\Throwable $ex) {
288:             throw new SmartException($ex->getMessage(), SmartException::ERR_CODE_JSON_PARSE_ERROR, SmartException::ERR_TYPE_PROGRAMMING_ERROR);
289:         }
290:     } // parseJsonInput

318:     /**
319:      * This is the main function that should be called to handle the API request.
320:      *
321:      * It detects the action, retrieves the request headers, parses the incoming JSON data and
322:      * tries to call the corresponding method for the API action.
323:      *
324:      * @return void
325:      *
326:      * @author Oleg Schildt
327:      */
328:     public function handleRequest()
329:     {
330:         try {
331:             $this->processInputData();
332: 
333:             $this->parseJsonInput();
334: 
335:             if (empty($this->action)) {
336:                 throw new SmartException("The action of the API request cannot be defined!", SmartException::ERR_CODE_SYSTEM_ERROR, SmartException::ERR_TYPE_PROGRAMMING_ERROR);
337:             }
338: 
339:             $robject = new \ReflectionObject($this);
340:             if (!$robject->hasMethod($this->action)) {
341:                 throw new SmartException(sprintf("No handler is defined for the API request action '%s'!", $this->action), SmartException::ERR_CODE_SYSTEM_ERROR, SmartException::ERR_TYPE_PROGRAMMING_ERROR);
342:             }
343: 
344:             $rmethod = $robject->getMethod($this->action);
345: 
346:             if ($this->action == "handleRequest" || $rmethod->isConstructor() || $rmethod->isDestructor() || !$rmethod->isPublic()) {
347:                 throw new SmartException(sprintf("The name '%s' for the API action is not supported!", $this->action), SmartException::ERR_CODE_SYSTEM_ERROR, SmartException::ERR_TYPE_PROGRAMMING_ERROR);
348:             }
349: 
350:             $this->response_data["result"] = "success";
351: 
352:             $this->preprocessRequest();
353:             $rmethod->invoke($this);
354:             $this->postprocessRequest();
355: 
356:             $this->sendJsonResponse();
357:         } catch (\Exception $ex) {
358:             $this->addExceptionError($ex);
359:             $this->exitWithErrors();
360:         }
361:     } // handleRequest
