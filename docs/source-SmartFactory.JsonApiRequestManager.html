116:     /**
117:      * Defines the current API request name.
118:      *
119:      * @return string
120:      * Returns the current API request.
121:      *
122:      * @author Oleg Schildt
123:      */
124:     public function getApiRequest()
125:     {
126:         if (empty($_SERVER['REQUEST_URI'])) {
127:             return "";
128:         }
129:         
130:         $api_request = $_SERVER['REQUEST_URI'];
131:         
132:         if (!empty($_SERVER['QUERY_STRING'])) {
133:             $api_request = str_replace($_SERVER['QUERY_STRING'], "", $api_request);
134:         }
135:         
136:         $api_request = rtrim($api_request, "/?");
137:         
138:         $api_base = str_replace(basename($_SERVER['SCRIPT_NAME']), "", $_SERVER['SCRIPT_NAME']);
139:         
140:         return str_replace($api_base, "", $api_request);
141:     } // getApiRequest

143:     /**
144:      * Registers a handler action for an API request call.
145:      *
146:      * @param string $api_request
147:      * The target API request.
148:      *
149:      * @param string $handler_class_name
150:      * The name of the class for handling this API request.
151:      *
152:      * Important! It should be a name of the class, neither the class instance
153:      * nor the class object. It is done to prevent situation that a wrong registration
154:      * of a handler breaks the handling of all requests.
155:      *
156:      * The class instantiating and class loading occurs only if this API request
157:      * comes.
158:      *
159:      * @return boolean
160:      * Returns true if the registration was successfull, otherwise false.
161:      *
162:      * @throws \Exception
163:      * It might throw the following exceptions in the case of any errors:
164:      *
165:      * - if the request name is not specified.
166:      * - if the request already has a handler.
167:      *
168:      * @see registerPreProcessHandler()
169:      * @see registerPostProcessHandler()
170:      * @see registerDefaultHandler()
171:      *
172:      * @author Oleg Schildt
173:      */
174:     public function registerApiRequestHandler($api_request, $handler_class_name)
175:     {
176:         if (empty($api_request)) {
177:             throw new \Exception("The API request is undefined (empty)!");
178:         }
179:         
180:         if (!empty($this->handler_table[$api_request])) {
181:             throw new \Exception("The API request '$api_request' has already the handler '" . $this->handler_table[$api_request] . "'!");
182:         }
183:         
184:         $this->handler_table[$api_request] = $handler_class_name;
185:         
186:         return true;
187:     } // registerApiRequestHandler

288:     /**
289:      * Handles an API request call trying to call the handler registered
290:      * for this API request.
291:      *
292:      * @return void.
293:      *
294:      * @throws \Exception
295:      * It might throw the following exceptions in the case of any errors:
296:      *
297:      * - if the creation of the handler fails.
298:      *
299:      * @uses IJsonApiRequestHandler
300:      *
301:      * @author Oleg Schildt
302:      */
303:     public function handleApiRequest()
304:     {
305:         $api_request = $this->getApiRequest();
306:         
307:         $response_data = [];
308:         $additional_headers = [];
309:         
310:         if (empty($api_request)) {
311:             $response_data["result"] = "error";
312:             
313:             $response_data["errors"] = [
314:                 ["error_code" => "system_error", "error_type" => "programming_error", "error_text" => "The API request is undefined (empty)!"]
315:             ];
316:             
317:             $this->sendJsonResponse($response_data, $additional_headers);
318:             
319:             return;
320:         }
321:         
322:         $handlers_to_call = [];
323:         
324:         if (!empty($this->handler_table["#pre_process#"])) {
325:             $handlers_to_call["#pre_process#"] = $this->handler_table["#pre_process#"];
326:         }
327:         
328:         if (empty($this->handler_table[$api_request])) {
329:             if (empty($this->handler_table["#default_handler#"])) {
330:                 $response_data["result"] = "error";
331:                 
332:                 $response_data["errors"] = [
333:                     [
334:                         "error_code" => "system_error", "error_type" => "programming_error",
335:                         "error_text" => sprintf("No handler is defined for the API request '%s'!", $api_request)
336:                     ]
337:                 ];
338:                 
339:                 $this->sendJsonResponse($response_data, $additional_headers);
340:                 
341:                 return;
342:             } else {
343:                 $handlers_to_call["#default_handler#"] = $this->handler_table["#default_handler#"];
344:             }
345:         } else {
346:             $handlers_to_call[$api_request] = $this->handler_table[$api_request];
347:         }
348:         
349:         if (!empty($this->handler_table["#post_process#"])) {
350:             $handlers_to_call["#post_process#"] = $this->handler_table["#post_process#"];
351:         }
352:         
353:         foreach ($handlers_to_call as $handler_key => $handler_class_name) {
354:             if (!class_exists($handler_class_name)) {
355:                 $response_data["result"] = "error";
356:                 
357:                 $response_data["errors"] = [
358:                     [
359:                         "error_code" => "system_error", "error_type" => "programming_error",
360:                         "error_text" => sprintf("The handler class '%s', defined for the request '%s', does not exist!", $handler_class_name, $handler_key)
361:                     ]
362:                 ];
363:                 
364:                 $this->sendJsonResponse($response_data, $additional_headers);
365:                 
366:                 return;
367:             }
368:             
369:             $handler_class = new \ReflectionClass($handler_class_name);
370:             
371:             if (!$handler_class->isSubclassOf("SmartFactory\Interfaces\IJsonApiRequestHandler")) {
372:                 $response_data["result"] = "error";
373:                 
374:                 $response_data["errors"] = [
375:                     [
376:                         "error_code" => "system_error", "error_type" => "programming_error",
377:                         "error_text" => sprintf("The handler class '%s', defined for the request '%s', does not implement the interface '%s'!", $handler_class_name, $handler_key, "IJsonApiRequestHandler")
378:                     ]
379:                 ];
380:                 
381:                 $this->sendJsonResponse($response_data, $additional_headers);
382:                 
383:                 return;
384:             }
385:             
386:             $handler = $handler_class->newInstance();
387:             
388:             $handler->handle($this, $api_request, $response_data, $additional_headers);
389:             
390:             if ($this->stop_handling) {
391:                 break;
392:             }
393:         } // foreach
394:         
395:         $this->sendJsonResponse($response_data, $additional_headers);
396:     } // handleApiRequest
397: } // JsonApiRequestManager
