 18:  *
 19:  * This function checks wether the request variable exists and

 21:  * an empty value is returned.
 22:  *
 23:  * @param string $name
 24:  * Name of the request variable how it is written in the form fields.
 25:  *
 26:  * @return mixed
 27:  * The value of the request variable or an empty string.
 28:  *
 29:  * @author Oleg Schildt

 31: function reqvar_value($name)
 32: {
 33:     if (empty($name)) {
 34:         return "";
 35:     }
 36:     
 37:     $name = preg_replace("/([^\[\]]+)(.*)/i", "\$_REQUEST[\\1]\\2", $name);
 38:     $name = preg_replace("/\[([^\[\]]+)\]/i", "[\"\\1\"]", $name);
 39:     $name = str_replace("[]", "", $name);
 40:     
 41:     $val = "";
 42:     // protection against the php injection
 43:     $name = preg_replace("/[;,\\(\\)\\?]+/", "", $name);
 44:     
 45:     eval("\$val = isset($name) ? $name : '';");
 46:     
 47:     return $val;
 48: } // reqvar_value

 59:  * - $parameters["id"] - id of the field.

 62:  *
 63:  * - $parameters["value"] - value that should be set for the field. If it is not specified,

 65:  *
 66:  * Any other attributes may be specified.
 67:  *
 68:  * Example:
 69:  * ```php

 74:  *             "placeholder" => "enter the data",
 75:  *             "title" => "enter the data",

 77:  *             "onblur" => "alert('Hello & Buy!')"
 78:  *            ]);
 79:  * ```
 80:  *
 81:  * @param boolean $echo
 82:  * If the value of $echo is true, the html code of the field is directly echoed.
 83:  * Otherwise it is returned as string. It might be useful to pass this code to
 84:  * other rendering functions instead of echoing.

 86:  * @return string|null
 87:  * If the paramter $echo is true, the string of the field html
 88:  * code is returned, otherwise null.

 99: function input_text($parameters, $echo = true)
100: {
101:     $val = "";
102:     
103:     $html = "<input type=\"text\"";
104:     
105:     $sys_fields = ["id", "name", "value", "formatter"];
106:     
107:     if (!empty($parameters["id"])) {
108:         $html .= " id=\"" . escape_html($parameters["id"]) . "\"";
109:     }
110:     if (!empty($parameters["name"])) {
111:         $html .= " name=\"" . escape_html($parameters["name"]) . "\"";
112:         $val = reqvar_value($parameters["name"]);
113:     }
114:     
115:     // Value setting logic:
116:     //
117:     // If explicitely set in value, than take it, otherwise take it from the request variable with that name.
118:     
119:     if (!empty($parameters["value"])) {
120:         $val = $parameters["value"];
121:     }
122:     
123:     if (!empty($parameters["formatter"]) && is_callable($parameters["formatter"])) {
124:         $val = call_user_func($parameters["formatter"], $val);
125:     }
126:     
127:     $html .= " value=\"" . escape_html($val) . "\"";
128:     
129:     foreach ($parameters as $name => $data) {
130:         if (in_array($name, $sys_fields)) {
131:             continue;
132:         }
133:         
134:         $html .= " " . escape_html($name) . "=\"" . escape_html($data) . "\"";
135:     }
136:     
137:     $html .= ">";
138:     
139:     if (!$echo) {
140:         return $html;
141:     }
142:     
143:     //echo("<pre>" . escape_html($html) . "</pre>");
144:     echo($html);
145:     
146:     return null;
147: } // input_text

497: function select($parameters, $echo = true)
498: {
499:     $val = "";
500:     
501:     $html = "<select";
502:     
503:     $sys_fields = ["id", "name", "value", "options", "multiple"];
504:     
505:     if (!empty($parameters["id"])) {
506:         $html .= " id=\"" . escape_html($parameters["id"]) . "\"";
507:     }
508:     if (!empty($parameters["name"])) {
509:         $html .= " name=\"" . escape_html($parameters["name"]) . "\"";
510:         $val = reqvar_value($parameters["name"]);
511:     }
512:     
513:     // Value setting logic:
514:     //
515:     // If explicitely set in value, than take it, otherwise take it from the request variable with that name.
516:     
517:     if (!empty($parameters["value"])) {
518:         $val = $parameters["value"];
519:     }
520:     
521:     foreach ($parameters as $name => $data) {
522:         if (in_array($name, $sys_fields)) {
523:             continue;
524:         }
525:         
526:         $html .= " " . escape_html($name) . "=\"" . escape_html($data) . "\"";
527:     }
528:     
529:     if (isset($parameters["multiple"])) {
530:         $html .= " multiple";
531:     }
532:     
533:     $html .= ">\n";
534:     
535:     if (!empty($parameters["options"])) {
536:         if (!is_array($parameters["options"])) {
537:             trigger_error("The parameter options must be an array val => text!", E_USER_ERROR);
538:         } else {
539:             foreach ($parameters["options"] as $optval => $text) {
540:                 $selected = "";
541:                 
542:                 if (!is_array($val) && $optval == $val) {
543:                     $selected = "selected";
544:                 }
545:                 if (is_array($val) && in_array($optval, $val)) {
546:                     $selected = "selected";
547:                 }
548:                 
549:                 $html .= "<option value=\"" . escape_html($optval) . "\" $selected>" . escape_html($text) . "</option>\n";
550:             }
551:         }
552:     }
553:     
554:     $html .= "</select>";
555:     
556:     if (!$echo) {
557:         return $html;
558:     }
559:     
560:     //echo("<pre>" . escape_html($html) . "</pre>");
561:     echo($html);
562:     
563:     return null;
564: } // select

616: function checkbox($parameters, $echo = true)
617: {
618:     $val = "";
619:     
620:     $html = "<input type=\"checkbox\"";
621:     
622:     $sys_fields = ["id", "name", "value", "checked"];
623:     
624:     if (!empty($parameters["id"])) {
625:         $html .= " id=\"" . escape_html($parameters["id"]) . "\"";
626:     }
627:     if (!empty($parameters["name"])) {
628:         $html .= " name=\"" . escape_html($parameters["name"]) . "\"";
629:         $val = reqvar_value($parameters["name"]);
630:     }
631:     
632:     if (!empty($parameters["value"])) {
633:         $html .= " value=\"" . escape_html($parameters["value"]) . "\"";
634:     }
635:     
636:     // Checked setting logic:
637:     //
638:     // If explicitely set in checked, than take it, otherwise take it from the request variable with that name.
639:

642:     if (!is_array($val) && !empty($val)) {
643:         $checked = " checked";
644:     }
645:     if (is_array($val) && in_array(checkempty($parameters["value"]), $val)) {
646:         $checked = " checked";
647:     }
648:     
649:     if (!empty($parameters["checked"])) {
650:         $checked = " checked";
651:     }
652:     
653:     $html .= $checked;
654:     
655:     foreach ($parameters as $name => $data) {
656:         if (in_array($name, $sys_fields)) {
657:             continue;
658:         }
659:         
660:         $html .= " " . escape_html($name) . "=\"" . escape_html($data) . "\"";
661:     }
662:

666:         return $html;
667:     }

678:  * This function renders a radiobutton field that is bound to
679:  * the request variable with the corresponding name.
680:  *
681:  * @param array $parameters
682:  * The array of parameters as an associative array in the form key => value:
683:  *
684:  * - $parameters["id"] - id of the field.
685:  *
686:  * - $parameters["name"] - name of the field.
687:  *
688:  * - $parameters["value"] - the value of the request variable that should be submitted
689:  * if the radiobutton is checked.
690:  *
691:  * - $parameters["checked"] - checked state that should be set for the field. If it is not specified,
692:  * the value of the corresponding request variable is used to set the checked state.
693:  *
694:  * Any other attributes may be specified.
695:  *
696:  * Example:
697:  * ```php
698:  * radiobutton(["id" => "data_radiocolor_red",
699:  *              "name" => "data[radiocolor]",
700:  *              "value" => "red"
701:  *             ]);
702:  *
703:  * radiobutton(["id" => "data_radiocolor_green",
704:  *              "name" => "data[radiocolor]",
705:  *              "value" => "green"
706:  *             ]);
707:  * ```
708:  *
709:  * @param boolean $echo
710:  * If the value of $echo is true, the html code of the field is directly echoed.
711:  * Otherwise it is returned as string. It might be useful to pass this code to
712:  * other rendering functions instead of echoing.
713:  *
714:  * @return string|null
715:  * If the paramter $echo is true, the string of the field html
716:  * code is returned, otherwise null.
717:  *
718:  * @see \SmartFactory\input_text()
719:  * @see \SmartFactory\input_password()
720:  * @see \SmartFactory\input_hidden()
721:  * @see \SmartFactory\textarea()
722:  * @see \SmartFactory\select()
723:  * @see \SmartFactory\checkbox()
724:  *
725:  * @author Oleg Schildt
726:  */
727: function radiobutton($parameters, $echo = true)
728: {
729:     $val = "";
730:     
731:     $html = "<input type=\"radio\"";
732:     
733:     $sys_fields = ["id", "name", "value", "checked"];
734:     
735:     if (!empty($parameters["id"])) {
736:         $html .= " id=\"" . escape_html($parameters["id"]) . "\"";
737:     }
738:     if (!empty($parameters["name"])) {
739:         $html .= " name=\"" . escape_html($parameters["name"]) . "\"";
740:         $val = reqvar_value($parameters["name"]);
741:     }
742:     
743:     if (!empty($parameters["value"])) {
744:         $html .= " value=\"" . escape_html($parameters["value"]) . "\"";
745:     }
746:     
747:     // Checked setting logic:
748:     //
749:     // If explicitely set in checked, than take it, otherwise take it from the request variable with that name.
750:     
751:     $checked = "";
752:     
753:     if (!empty($val) && checkempty($parameters["value"]) == $val) {
754:         $checked = " checked";
755:     }
756:     
757:     if (!empty($parameters["checked"])) {
758:         $checked = " checked";
759:     }
760:     
761:     $html .= $checked;
762:     
763:     foreach ($parameters as $name => $data) {
764:         if (in_array($name, $sys_fields)) {
765:             continue;
766:         }
767:         
768:         $html .= " " . escape_html($name) . "=\"" . escape_html($data) . "\"";
769:     }
770:     
771:     $html .= ">";
772:     
773:     if (!$echo) {
774:         return $html;
775:     }
776:     
777:     //echo("<pre>" . escape_html($html) . "</pre>");
778:     echo($html);
779:     
780:     return null;

783: /**
784:  * Function for rendering tables.

786:  * This function renders a tables based on the values of an array.
787:  *
788:  * @param array &$array
789:  * The array of the rows.

791:  * @param array $parameters

793:  *
794:  * - $parameters["captions"] - the associative array of column captions. If
795:  * it is not specified, the table is rendered without captions.
796:  *
797:  * - $parameters["col_class_from_keys"] - if it is specified and set to true,
798:  * the keys of the captions are used as the class names.
799:  *
800:  * - $parameters["no_escape_html"] - if it is specified and set to true,
801:  * the values of the data array are not escaped and passed as is. It is useful
802:  * when html codes of other elements are passed. Per default, the table data
803:  * is escaped for html.
804:  *
805:  * - $parameters["formatter"] - this function is used for the value formatting
806:  * if specified. It is called for evelry cell of the table. The signature of
807:  * this function is:
808:  *
809:  * ```php
810:  * function ($rownum, $colnum, $colname, $val) : string;
811:  * ```
812:  *
813:  * - $rownum - the row number.
814:  *
815:  * - $colnum - the colnum number.
816:  *
817:  * - $colname - the column name if the $parameters["captions"] are provided.
818:  * The keys of the cpation array are used. If the $parameters["captions"]
819:  * are not provided, $colname is equal to $colnum.
820:  *
821:  * - $val - original value of the table cell.
822:  *
823:  * If necessary, format the original value and return it. Otherwise,
824:  * return the original value without changes.
825:  *
826:  * Any other attributes may be specified.
827:  *
828:  * Example:
829:  * ```php
830:  * $rows = [
831:  *    ["name" => "DB design", "employee" => "Alex", ... ],
832:  *    ["name" => "Mask implementation", "employee" => "Boris", ... ],
833:  *    ["name" => "Validation", "employee" => "Alex", ... ],
834:  *    ["name" => "Settings", "employee" => "Boris", ... ],
835:  *    ["name" => "About", "employee" => "Robert", ... ],
836:  *    ["name" => "Initialization", "employee" => "Alon", ... ]
837:  * ];
838:  *
839:  * $captions = [
840:  *   "name" => "Task name",
841:  *   "employee" => "Employee",
842:  *   "time_estimation" => "Time estimation",
843:  *   "deadline" => "Deadline",
844:  *   "comments" => "Conmments"
845:  * ];
846:  *
847:  * $formatter = function ($rownum, $colnum, $colname, $val) {
848:  *   if($colname == "time_estimation") return format_number($val, 2);
849:  *   if($colname == "deadline") return date("Y-m-d H:i", $val);
850:  *   return $val;
851:  * };
852:  *
853:  * table($rows,
854:  *       ["captions" => $captions,
855:  *        "class" => "my_table",
856:  *        "style" => "background-color: #dddddd",
857:  *        "col_class_from_keys" => true,
858:  *        "no_escape_html" => false,
859:  *        "formatter" => $formatter
860:  *       ]);
861:  * ```
862:  *
863:  * @param boolean $echo
864:  * If the value of $echo is true, the html code of the table is directly echoed.
865:  * Otherwise it is returned as string. It might be useful to pass this code to
866:  * other rendering functions instead of echoing.
867:  *
868:  * @return string|null
869:  * If the paramter $echo is true, the string of the table html
870:  * code is returned, otherwise null.
871:  *
872:  * @author Oleg Schildt
873:  */
