 56:     /**
 57:      * This is internal auxiliary function for preparing messags for displaying.
 58:      *
 59:      * When the messages are requested, it is assumed they will be displayed. Thus,
 60:      * the message array is cleared to avoid displaying of the same messages twice.
 61:      *
 62:      * The message details are removed if the flag show_message_details is false.
 63:      *
 64:      * @param array $messages
 65:      * The source array of messages.
 66:      *
 67:      * @return array
 68:      * Returns the array of messages for displaying.
 69:      *
 70:      * @author Oleg Schildt
 71:      */
 72:     protected function extractMessagesForDisplay(&$messages)
 73:     {
 74:         $output = [];
 75:         
 76:         if (empty($messages) || count($messages) == 0) {
 77:             return $output;
 78:         }
 79:         
 80:         foreach ($messages as $current) {
 81:             $message_entry["message"] = $current["message"];
 82:             
 83:             if (!empty($current["code"])) {
 84:                 $message_entry["code"] = $current["code"];
 85:             }
 86:             
 87:             if (!empty($current["auto_hide"])) {
 88:                 $message_entry["auto_hide"] = $this->getAutoHideTime();
 89:             }
 90:             
 91:             if (!empty($current["details"]) && config_settings()->getParameter("show_message_details", false, 1)) {
 92:                 $message_entry["details"] = $current["details"];
 93:             }
 94:             
 95:             $output[] = $message_entry;
 96:         } // foreach
 97:         
 98:         // unset messages because they has been shown
 99:         $messages = null;
100:         
101:         return $output;
102:     } // extractMessagesForDisplay
