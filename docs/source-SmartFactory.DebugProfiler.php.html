 53:     /**
 54:      * Initializes the debug profiler with parameters.
 55:      *
 56:      * @param array $parameters
 57:      * Settings for logging as an associative array in the form key => value:
 58:      *
 59:      * - $parameters["log_path"] - the target file path where the logs should be stored.
 60:      *
 61:      * - $parameters["write_source_file_and_line_by_debug"] - the flag that defines whether for each debug output the source file and
 62:      * line number are written from where it is called.
 63:      *
 64:      * @return void
 65:      *
 66:      * @throws \Exception
 67:      * It might throw an exception in the case of any system errors:
 68:      *
 69:      * - if the log path is not specified.
 70:      * - if the trace file is not writable.
 71:      *
 72:      * @author Oleg Schildt
 73:      */
 74:     public function init($parameters)
 75:     {
 76:         $this->write_source_file_and_line_by_debug = !empty($parameters["write_source_file_and_line_by_debug"]);
 77: 
 78:         if (empty($parameters["log_path"])) {
 79:             throw new \Exception("Log path is not specified!");
 80:         }
 81:     
 82:         $this->log_path = rtrim(str_replace("\\", "/", $parameters["log_path"]), "/") . "/";
 83:         
 84:         if (!file_exists($this->log_path) || !is_writable($this->log_path)) {
 85:             throw new \Exception(sprintf("The log path '%s' is not writable!", $this->log_path));

 89:     /**
 90:      * Extracts the call stack from the standard PHP backtrace (debug_backtrace).
 91:      *
 92:      * @param array $btrace
 93:      * The backtrace.
 94:      *
 95:      * @return string
 96:      * Returns the extracted call stack.
 97:      *
 98:      * @author Oleg Schildt
 99:      */
100:     protected function extract_call_stack($btrace)
101:     {
102:         if (empty($btrace) || !is_array($btrace)) {
103:             return "";
104:         }
105: 
106:         $trace = "";
107: 
108:         $indent = "";
109:         foreach ($btrace as $btrace_entry) {
110: 
111:             if (!empty($btrace_entry["function"]) && ($btrace_entry["function"] == "handle_error" || strpos($btrace_entry["function"], "{closure}") !== false || $btrace_entry["function"] == "handleError" || $btrace_entry["function"] == "trigger_error")) {
112:                 continue;
113:             }
114: 
115:             if (empty($btrace_entry["file"])) {
116:                 continue;
117:             }
118: 
119:             if (!empty($btrace_entry["function"])) {
120:                 $trace .= $indent . str_replace("SmartFactory\\", "", $btrace_entry["function"]) . "() ";
121:             }
122: 
123:             $trace .= "[";
124: 
125:             $trace .= str_replace("\\", "/", $btrace_entry["file"]);
126: 
127:             $trace .= ", ";
128: 
129:             if (empty($btrace_entry["line"])) {
130:                 $trace .= "line number undefined";
131:             } else {
132:                 $trace .= $btrace_entry["line"];
133:             }
134: 
135:             $trace .= "]";
136: 
137:             $trace .= "\r\n";
138: 
139:             $indent .= "  ";
140:         }
141: 
142:         return trim($trace);
143:     } // extract_call_stack

145:     /**
146:      * Logs a message to a standard debug output (logs/debug.log).
147:      *
148:      * @param string $message
149:      * Message to be logged.
150:      *
151:      * @param string $file_name
152:      * The target file name.
153:      *
154:      * @param boolean $show_callstack
155:      * The flag that defines whether the callstack should be also traced.
156:      *
157:      * @return boolean
158:      * It should return true if the logging was successful, otherwise false.
159:      *
160:      * @throws \Exception
161:      * It might throw an exception in the case of any errors:
162:      *
163:      * - if the debug file is not writable.
164:      *
165:      * @author Oleg Schildt
166:      */
167:     public function debugMessage($message, $file_name = "debug.log", $show_callstack = false)
168:     {
169:         $logfile = $this->log_path . $file_name;
170: 
171:         $backfiles = debug_backtrace();
172: 
173:         if (basename($backfiles[0]['file']) == "short_functions_inc.php" && $backfiles[0]['function'] == "debugMessage") {
174:             $file = empty($backfiles[1]['file']) ? "" : $backfiles[1]['file'];
175:             $line = empty($backfiles[1]['line']) ? "" : $backfiles[1]['line'];
176:         } else {
177:             $file = empty($backfiles[0]['file']) ? "" : $backfiles[0]['file'];
178:             $line = empty($backfiles[0]['line']) ? "" : $backfiles[0]['line'];
179:         }
180: 
181:         $message = trim($message);
182: 
183:         if($this->write_source_file_and_line_by_debug && !empty($file) && !empty($line)) {
184:             $message = "\r\n#source: " . $file . ", " . $line . "\r\n\r\n" . $message;
185:         }
186: 
187:         $message .= "\r\n";
188: 
189:         if ($show_callstack) {
190:             $message .= "\r\n" . $this->extract_call_stack($backfiles) . "\r\n";
191:         }
192: 
193:         if ((!file_exists($logfile) && is_writable($this->log_path)) || is_writable($logfile)) {
194:             return file_put_contents($logfile, $message, FILE_APPEND) !== false;
195:         } else {
196:             throw new \Exception(sprintf("The log file '%s' is not writable!", $logfile));
197:         }
198:     } // debugMessage

285:     /**

287:      *
288:      * @return boolean
289:      * It should return true if the files have been successfully deleted, otherwise false.

291:      * @see DebugProfiler::clearLogFile()

296:     {
297:         $dir = $this->log_path;
298:         $files = scandir($dir);
299:         foreach ($files as $file) {
300:             if ($file == "." || $file == ".." || is_dir($dir . $file)) {
301:                 continue;
302:             }
303:             
304:             $pi = pathinfo($dir . $file);
305:             
306:             if (empty($pi["extension"]) || strtolower($pi["extension"]) != "log") {
307:                 continue;
308:             }
309:             
310:             if (!$this->clearLogFile($file)) {
311:                 return false;
312:             }
313:         }
314:         
315:         return true;
