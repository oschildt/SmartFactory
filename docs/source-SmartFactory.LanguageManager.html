146:     public function loadDictionary()
147:     {
148:         if (self::$dictionary_loaded) {
149:             return true;
150:         }
151:         
152:         if ($this->use_apcu) {
153:             do {
154:                 if (!apcu_exists("dictionary_supported_languages")) {
155:                     break;
156:                 }
157:                 
158:                 self::$supported_languages = apcu_fetch("dictionary_supported_languages");
159:                 if (empty(self::$supported_languages)) {
160:                     break;
161:                 }
162:                 
163:                 if (!apcu_exists("dictionary_languages")) {
164:                     break;
165:                 }
166:                 self::$languages = apcu_fetch("dictionary_languages");
167:                 if (empty(self::$languages)) {
168:                     break;
169:                 }
170:                 
171:                 if (!apcu_exists("dictionary_countries")) {
172:                     break;
173:                 }
174:                 self::$countries = apcu_fetch("dictionary_countries");
175:                 if (empty(self::$countries)) {
176:                     break;
177:                 }
178:                 
179:                 if (!apcu_exists("dictionary_texts")) {
180:                     break;
181:                 }
182:                 self::$texts = apcu_fetch("dictionary_texts");
183:                 if (empty(self::$texts)) {
184:                     break;
185:                 }
186:                 
187:                 return true;
188:             } while (false);
189:         }
190:         
191:         $json = file_get_contents($this->localization_path . "texts.json");
192:         if ($json === false) {
193:             throw new \Exception("Translation file '" . $this->localization_path . "texts.json" . "' cannot be loaded or does not exist!");
194:         }
195:         
196:         $json_array = [];
197:         try {
198:             json_to_array($json, $json_array);
199:         } catch (\Throwable $ex) {
200:             throw new \Exception("Translation file '" . $this->localization_path . "texts.json" . "' is invalid!" . "\n\n" . $ex->getMessage());
201:         }
202:         
203:         if (!empty($json_array["interface_languages"])) {
204:             foreach ($json_array["interface_languages"] as $lang_code) {
205:                 self::$supported_languages[$lang_code] = $lang_code;
206:             }
207:         }
208:         
209:         if (!empty($json_array["languages"])) {
210:             foreach ($json_array["languages"] as $text_id => &$translations) {
211:                 foreach ($translations as $lang_code => $translation) {
212:                     self::$languages[$lang_code][$text_id] = $translation;
213:                 }
214:             }
215:         }
216:         
217:         if (!empty($json_array["countries"])) {
218:             foreach ($json_array["countries"] as $text_id => &$translations) {
219:                 foreach ($translations as $lang_code => $translation) {
220:                     self::$countries[$lang_code][$text_id] = $translation;
221:                 }
222:             }
223:         }
224:         
225:         if (!empty($json_array["texts"])) {
226:             foreach ($json_array["texts"] as $text_id => &$translations) {
227:                 foreach ($translations as $lang_code => $translation) {
228:                     self::$texts[$lang_code][$text_id] = $translation;
229:                 }
230:             }
231:         }
232:         
233:         self::$dictionary_loaded = true;
234:         
235:         foreach (self::$supported_languages as $lng) {
236:             self::$current_language = $lng;
237:             break;
238:         }
239:         
240:         if ($this->use_apcu) {
241:             apcu_store("dictionary_supported_languages", self::$supported_languages);
242:             apcu_store("dictionary_languages", self::$languages);
243:             apcu_store("dictionary_countries", self::$countries);
244:             apcu_store("dictionary_texts", self::$texts);
245:         }
246:         
247:         return true;
248:     } // loadDictionary

250:     /**
251:      * This function should detect the current language based on cookies, browser languages etc.
252:      *
253:      * Priority:
254:      *
255:      * 1. explicitly set by the request parameter language.
256:      * 2. last language in the session.
257:      * 3. last language in the cookie.
258:      * 4. browser default language.
259:      * 5. the first one from the supported list.
260:      * 6. English.
261:      *
262:      * @param string $context
263:      * The context of the application.
264:      *
265:      * Some applications may consist of two parts - administration
266:      * console and public site. A usual example is a CMS system.
267:      *
268:      * For example, you are using administration console in English
269:      * and editing the public site for German and French.
270:      * When you open the public site for preview in German or French,
271:      * you want it to be open in the corresponding language, but
272:      * the administration console should remain in English.
273:      *
274:      * With the help of $context, you are able to maintain different
275:      * languages for different parts of your application.
276:      * If you do not need the $context, just do not specify it.
277:      *
278:      * @return void
279:      *
280:      * @author Oleg Schildt
281:      */
282:     public function detectLanguage($context = "default")
283:     {
284:         self::$context = $context;
285:         
286:         // Priority:
287:         
288:         // 1) explicitly set by the request parameter language
289:         // 2) last language in the session
290:         // 3) last language in the cookie
291:         // 4) browser default language
292:         // 5) the first one from the supported list
293:         // 6) English
294:         
295:         // Let's go
296:         
297:         $language = "en";
298:         
299:         // 5) the first one from the supported list
300:         foreach (self::$supported_languages as $lng) {
301:             $language = $lng;
302:             break;
303:         }
304:         
305:         // 4) browser default
306:         if (isset($_SERVER["HTTP_ACCEPT_LANGUAGE"]) && trim($_SERVER["HTTP_ACCEPT_LANGUAGE"]) != "") {
307:             $accepted = explode(',', $_SERVER["HTTP_ACCEPT_LANGUAGE"]);
308:             
309:             foreach ($accepted as $key => $name) {
310:                 $code = explode(';', $name);
311:                 // handle the cases like en-ca => en
312:                 $code = explode("-", $code[0]);
313:                 
314:                 if (!empty(self::$supported_languages[$code[0]])) {
315:                     $language = $code[0];
316:                     break;
317:                 }
318:             }
319:         }
320:         
321:         // 3) last language in the cookie
322:         if (!empty($_COOKIE[self::$context . "_language"]) && !empty(self::$supported_languages[$_COOKIE[self::$context . "_language"]])) {
323:             $language = $_COOKIE[self::$context . "_language"];
324:         }
325:         
326:         // 2) last language in the session
327:         if (!empty(session()->vars()[self::$context . "_language"]) && !empty(self::$supported_languages[session()->vars()[self::$context . "_language"]])) {
328:             $language = $_COOKIE[self::$context . "_language"];
329:         }
330:         
331:         // 1) explicitly set by request parameter language
332:         if (!empty($_REQUEST["language"]) && !empty(self::$supported_languages[$_REQUEST["language"]])) {
333:             $language = $_REQUEST["language"];
334:         }
335:         
336:         $this->setCurrentLanguage($language);
337:     } // detectLanguage
