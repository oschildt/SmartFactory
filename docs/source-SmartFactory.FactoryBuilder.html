 13: /**
 14:  * Class for creation of the objects.
 15:  *
 16:  * The class FactoryBuilder is an auxiliary class that empowers the factory methods. It provides the methods
 17:  * for binding class implementations to the interfaces.
 18:  *
 19:  * @author Oleg Schildt
 20:  */
 21: class FactoryBuilder
 22: {
 23:     /**
 24:      * Internal array for storing the singleton instances for re-using.
 25:      *
 26:      * @var array
 27:      *
 28:      * @author Oleg Schildt
 29:      */
 30:     static protected $singletons = [];
 31:     
 32:     /**
 33:      * Internal array for storing the mapping betwen interfaces/classes and the bound classes.
 34:      *
 35:      * @var array
 36:      *
 37:      * @author Oleg Schildt
 38:      */
 39:     static protected $itable = [];
 40:     
 41:     /**
 42:      * Binds a class to an interface or a parent class.
 43:      *
 44:      * The key point of this approach is the definition common interfaces
 45:      * and implementation of them in classes. When an object that supoorts an
 46:      * interface is requested, an instance of the corresponding bound class
 47:      * is created. When you want to change the class, you need just bind
 48:      * the new class to the interface.
 49:      *
 50:      * You can also bind a class to itself and reqest it by own name in factory
 51:      * method. And later, you can implement a derived class and change the binding
 52:      * without affecting the code in the business logic.
 53:      *
 54:      * @param string|object $interface_or_class
 55:      * Name of the class/interface as string or the class/interface.
 56:      *
 57:      * @param string|object $class
 58:      * The class which instance should be created if the object is requested.
 59:      *
 60:      * @param callable $init_function
 61:      * The optional initialization function. You can provide it to do some
 62:      * custom intialization. The signature of
 63:      * this function is:
 64:      *
 65:      * ```php
 66:      * function (object $instance) : void;
 67:      * ```
 68:      *
 69:      * - $instance - the created instance.
 70:      *
 71:      * Example:
 72:      *
 73:      * ```php
 74:      * FactoryBuilder::bindClass(ILanguageManager::class, LanguageManager::class, function($instance) {
 75:      *   $instance->detectLanguage();
 76:      * });
 77:      *
 78:      * FactoryBuilder::bindClass(IRecordsetManager::class, RecordsetManager::class, function($instance) {
 79:      *   $instance->setDBWorker(dbworker());
 80:      * });
 81:      * ```
 82:      *
 83:      * @return void
 84:      *
 85:      * @throws \Exception
 86:      * It might throw the following exceptions in the case of any errors:
 87:      *
 88:      * - if the interface or bound class is not specified.
 89:      * - if the interface or class does not exist.
 90:      * - if the bound class is empty.
 91:      * - if the bound class does not implement the corresponding interface.
 92:      * - if the bound class is not instantiable.
 93:      * - if the check of the classes and interfaces fails.
 94:      *
 95:      * @author Oleg Schildt
 96:      */
 97:     static public function bindClass($interface_or_class, $class, $init_function = null)
 98:     {
 99:         if (empty($class)) {
100:             throw new \Exception("Bound class is empty!");
101:         }
102:         
103:         if (empty($interface_or_class)) {
104:             throw new \Exception("Bound interface or class is empty!");
105:         }
106:         
107:         if (!interface_exists($interface_or_class) && !class_exists($interface_or_class)) {
108:             throw new \Exception(sprintf("The interface or class '%s' does not exist!", $interface_or_class));
109:         }
110:         
111:         if (!class_exists($class)) {
112:             throw new \Exception(sprintf("The class '%s' does not exist!", $class));
113:         }
114:         
115:         try {
116:             $ic = new \ReflectionClass($interface_or_class);
117:             $c = new \ReflectionClass($class);
118:         } catch (\Exception $ex) {
119:             throw new \Exception($ex->getMessage());
120:         }
121:         
122:         if (!$c->isInstantiable()) {
123:             throw new \Exception(sprintf("The class '%s' is not instantiable!", $c->getName()));
124:         }
125:         
126:         if ($c != $ic) {
127:             if (!$c->isSubclassOf($ic)) {
128:                 throw new \Exception(sprintf("The class '%s' does not implement the interface '%s'!", $c->getName(), $ic->getName()));
129:             }
130:         }
131:         
132:         $f = null;
133:         if ($init_function !== null) {
134:             if (!is_callable($init_function)) {
135:                 throw new \Exception(sprintf("'%s' is not a function!", $init_function));
136:             }
137:             
138:             try {
139:                 $f = new \ReflectionFunction($init_function);
140:             } catch (\Exception $ex) {
141:                 throw new \Exception($ex->getMessage());
142:             }
143:         }
144:         
145:         self::$itable[$ic->getName()] = ["class" => $c, "init_function" => $f];
146:     } // bindClass
147:     
148:     /**
149:      * Сreates an object that support the interface $interface_or_class.
150:      *
151:      * @param string|object $interface_or_class
152:      * Name of the class/interface which instance should be created.
153:      *
154:      * @param boolean $singleton
155:      * If the parameter is true, it ensures that only one instance of this object exists.
156:      * The singleton is a usual patter for the action objects like SessionManager, EventManager,
157:      * DBWorker etc. It makes no sense to produce many instances of such classes,
158:      * it wastes the computer resources and might cause errors.
159:      *
160:      * If the parameter is false, then, by each request, a new object is created. If you request
161:      * data objects like User, a separate instance must be created for each item.
162:      *
163:      * @return object
164:      * Returns object of the class bound to the interface.
165:      *
166:      * @throws \Exception
167:      * It might throw the following exceptions in the case of any errors:
168:      *
169:      * - if the interface or class is not specified.
170:      * - if the interface or class does not exist.
171:      * - if the check of the classes and interfaces fails.
172:      *
173:      * @used_by \SmartFactory\instance()
174:      * @used_by \SmartFactory\singleton()
175:      *
176:      * @author Oleg Schildt
177:      */
178:     static public function getInstance($interface_or_class, $singleton)
179:     {
180:         if (empty($interface_or_class)) {
181:             throw new \Exception("Class or interface is not specified!");
182:         }
183:         
184:         if (!interface_exists($interface_or_class) && !class_exists($interface_or_class)) {
185:             throw new \Exception(sprintf("The interface or class '%s' does not exist!", $interface_or_class));
186:         }
187:         
188:         try {
189:             $class = new \ReflectionClass($interface_or_class);
190:         } catch (\Exception $ex) {
191:             throw new \Exception($ex->getMessage());
192:         }
193:         
194:         $class_name = $class->getName();
195:         
196:         if (empty(self::$itable[$class_name])) {
197:             throw new \Exception(sprintf("The interface or class '%s' has no bound class!", $class_name));
198:         }
199:         
200:         try {
201:             // if not singleton, we create a new instance every time it is requested
202:             if (!$singleton) {
203:                 $instance = self::$itable[$class_name]["class"]->newInstance();
204:                 
205:                 if (!empty(self::$itable[$class_name]["init_function"])) {
206:                     self::$itable[$class_name]["init_function"]->invoke($instance);
207:                 }
208:                 
209:                 return $instance;
210:             }
211:             
212:             // if singleton, we create an instance only if it does not exist yet
213:             if (empty(self::$singletons[$class_name])) {
214:                 self::$singletons[$class_name] = self::$itable[$class_name]["class"]->newInstance();
215:                 
216:                 if (!empty(self::$itable[$class_name]["init_function"])) {
217:                     self::$itable[$class_name]["init_function"]->invoke(self::$singletons[$class_name]);
218:                 }
219:             }
220:         } catch (\Exception $ex) {
221:             throw new \Exception($ex->getMessage());
222:         }
223:         
224:         return self::$singletons[$class_name];
225:     } // getInstance
226: } // FactoryBuilder
