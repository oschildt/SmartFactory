 15: /**
 16:  * Class for managing connections to the DB shards.
 17:  *
 18:  * This shard manager allows registering multiple shards and request connections to them.
 19:  * It ensures that only one connection to a shard is used within a request.
 20:  *
 21:  * @see DBWorker
 22:  *
 23:  * @author Oleg Schildt
 24:  */
 25: class ShardManager implements IShardManager
 26: {
 27:     /**
 28:      * Internal array for storing the load balancing groups and its shards.
 29:      *
 30:      * @var array
 31:      *
 32:      * @author Oleg Schildt
 33:      */
 34:     protected $load_balancing_groups = [];
 35:     
 36:     /**
 37:      * Internal array for storing the mapping betwen shard names and connection parameters.
 38:      *
 39:      * @var array
 40:      *
 41:      * @author Oleg Schildt
 42:      */
 43:     protected $shard_table = [];
 44:     
 45:     /**
 46:      * Registers a new shard.
 47:      *
 48:      * @param string $shard_name
 49:      * Unique shard name.
 50:      *
 51:      * @param array $parameters
 52:      * The connection parameters to the shard as an associative array in the form key => value:
 53:      *
 54:      * - $parameters["db_type"] - type of the database (MySQL or MSSQL)
 55:      *
 56:      * - $parameters["db_server"] - server address
 57:      *
 58:      * - $parameters["db_name"] - database name
 59:      *
 60:      * - $parameters["db_user"] - user name
 61:      *
 62:      * - $parameters["db_password"] - user password
 63:      *
 64:      * - $parameters["autoconnect"] - should true if the connection should be established automatically
 65:      *                 upon creation.
 66:      *
 67:      * - $parameters["read_only"] - this paramter sets the connection to the read only mode.
 68:      *
 69:      * @param string $load_balancing_group
 70:      * The name of the load balancing group, if the shard should be part of it, {@see ShardManager::randomDBShard()}.
 71:      *
 72:      * @return boolean
 73:      * It should return true if the registering was successful, otherwise false.
 74:      *
 75:      * @throws \Exception
 76:      * It might throw an exception in the case of any errors:
 77:      *
 78:      * - if the shard name is not specified.
 79:      * - if the shard name has been already registered.
 80:      *
 81:      * @author Oleg Schildt
 82:      */
 83:     public function registerShard($shard_name, $parameters, $load_balancing_group = "")
 84:     {
 85:         if (empty($shard_name)) {
 86:             throw new \Exception("The shard name is not specified!");
 87:         }
 88:         
 89:         if (!empty($this->shard_table[$shard_name])) {
 90:             throw new \Exception("The shard '$shard_name' has been already registered!");
 91:         }
 92:         
 93:         $this->shard_table[$shard_name]["parameters"] = $parameters;
 94:         
 95:         if (!empty($load_balancing_group)) {
 96:             $this->load_balancing_groups[$load_balancing_group][] = $shard_name;
 97:         }
 98:         
 99:         return true;
100:     } // registerShard
101:     
102:     /**
103:      * The method provides the DBWorker object for working with the shard.
104:      *
105:      * If the parameters are omitted, the system takes the parameters from the configuration
106:      * settings and reuses the single instance of the DBWorker for all requests.
107:      * If the user passes the parameters explicitly, a new instance of the DBWorker is created upon each new request.
108:      *
109:      * Currently supported: MySQL und MS SQL.
110:      *
111:      * @param string $shard_name
112:      * The name of the shard.
113:      *
114:      * @return \SmartFactory\DatabaseWorkers\DBWorker|null
115:      * returns DBWorker object or null if the object could not be created.
116:      *
117:      * @throws \Exception
118:      * It might throw an exception in the case of any errors:
119:      *
120:      * - if the interface or class does not exist.
121:      * - if the shard was not found.
122:      * - if the check of the classes and interfaces fails.
123:      * - if the php extension is not installed.
124:      * - db_missing_type_error - if the database type is not specified.
125:      * - db_conn_data_error - if the connection parameters are incomplete.
126:      * - db_server_conn_error - if the database server cannot be connected.
127:      * - db_not_exists_error - if database does not exists od inaccesible to the user.
128:      *
129:      * @author Oleg Schildt
130:      */
131:     public function dbshard($shard_name)
132:     {
133:         if (empty($shard_name) || empty($this->shard_table[$shard_name])) {
134:             throw new \Exception("The shard '$shard_name' was not found!");
135:             return null;
136:         }
137:         
138:         // Important!
139:         //
140:         // Many different databases may be connected within one request,
141:         // therefore, we request the dbworker as not singleton. Otherwise,
142:         // only one dbworker  connected to one database would be returned.
143:         //
144:         // But, as soon as, we got a shard connected to a required database,
145:         // we keep it as singleton within one request, to avoid the unnecessary
146:         // DB connections.
147:         //
148:         // In other words, many databases can be accessed from one request,
149:         // but there is only one connection to each database.
150:         
151:         if (empty($this->shard_table[$shard_name]["dbworker"])) {
152:             $this->shard_table[$shard_name]["dbworker"] = dbworker($this->shard_table[$shard_name]["parameters"], false /* not singleton */);
153:         }
154:         
155:         return $this->shard_table[$shard_name]["dbworker"];
156:     } // dbshard
157:     
158:     /**
159:      * The method provides the DBWorker object for working with the shard, that is chosen randomly
160:      * for load balancing reason.
161:      *
162:      * @param string $load_balancing_group
163:      * The name of the load balancing group, from which the shard should be randomly picked.
164:      *
165:      * @return \SmartFactory\DatabaseWorkers\DBWorker|null
166:      * returns DBWorker object or null if the object could not be created.
167:      *
168:      * @throws \Exception
169:      * It might throw an exception in the case of any errors:
170:      *
171:      * - if the load balancing group was not found.
172:      * - db_server_conn_error - if the database server cannot be connected.
173:      * - db_not_exists_error - if database does not exists od inaccesible to the user.
174:      *
175:      * @author Oleg Schildt
176:      */
177:     public function randomDBShard($load_balancing_group)
178:     {
179:         if (empty($this->load_balancing_groups[$load_balancing_group])) {
180:             throw new \Exception("The group '$load_balancing_group' was not found!");
181:         }
182:         
183:         $pos = rand(0, count($this->load_balancing_groups[$load_balancing_group]) - 1);
184:         
185:         return $this->dbshard($this->load_balancing_groups[$load_balancing_group][$pos]);
186:     } // randomDBShard
187: } // ShardManager
