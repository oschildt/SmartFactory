135: /**
136:  * Short function for reporting the SQL error.
137:  *
138:  * It should be called if a SQL query failed.
139:  *
140:  * @param DatabaseWorkers\DBWorker $dbw
141:  * The dbworker used to perform the SQL query.
142:  *
143:  * @return false
144:  * Returns always false.
145:  *
146:  * Example:
147:  *
148:  * ```php
149:  * function some_function()
150:  * {
151:  *    ...
152:  *    if(!dbworker()->execute_query("SELECT FIRST_NAME, LAST_NAME FROM USERS"))
153:  *    {
154:  *      return sql_error(dbworker());
155:  *    }
156:  *    ...
157:  * }
158:  * ```
159:  */
160: function sql_error($dbw)
161: {
162:   messenger()->setError(text("ErrQueryFailed"),
163:                         $dbw->get_last_error() . "\n\n" .
164:                         $dbw->get_last_query()
165:                        );
166:   return false;
167: } // sql_error
