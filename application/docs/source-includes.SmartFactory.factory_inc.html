 24: /**
 25:  * The method singleton creates an object that support the specified interface and ensures 
 26:  * that only one instance of this object exists.
 27:  *
 28:  * The singleton is a usual pattern for the action objects like SessionManager, EventManager, 
 29:  * DBWorker etc. It makes no sense to produce many instances of such classes, 
 30:  * it wastes the computer resources and might cause errors.
 31:  *
 32:  * @param string|object $interface_or_class 
 33:  * Name of the class/interface as string or the class/interface.
 34:  *
 35:  * @return object
 36:  * Returns object of the class bound to the interfase.
 37:  *
 38:  * @throws \Exception
 39:  * - If the interface or class is not specified.
 40:  * - If the interface or class does not exist.
 41:  * - If the interface or class has no bound class.
 42:  *
 43:  * @see \SmartFactory\instance()
 44:  *
 45:  * @author Oleg Schildt
 46:  */
 47: function singleton($interface_or_class)
 48: {
 49:   return FactoryBuilder::getInstance($interface_or_class, true);
 50: } // singleton
 51: 
 52: /**
 53:  * The method instance creates an object that support the specified interface. 
 54:  * 
 55:  * By each request, a new object is created. If you request data objects like User, 
 56:  * a separate instance must be created for each item.
 57:  *
 58:  * @param string|object $interface_or_class 
 59:  * Name of the class/interface as string or the class/interface.
 60:  *
 61:  * @return object
 62:  * Returns object of the class bound to the interface.
 63:  *
 64:  * @throws \Exception
 65:  * - If the interface or class is not specified.
 66:  * - If the interface or class does not exist.
 67:  * - If the interface or class has no bound class.
 68:  *
 69:  * @see \SmartFactory\singleton()
 70:  *
 71:  * @author Oleg Schildt
 72:  */
 73: function instance($interface_or_class)
 74: {
 75:   return FactoryBuilder::getInstance($interface_or_class, false);
 76: } // instance

 78: /**
 79:  * The method dbworker provides the DBWorker object for working with the database. 
 80:  * 
 81:  * If the parameters are omitted, the system takes the parameters from the configuration 
 82:  * settings and reuses the single instance of the DBWorker for all requests. 
 83:  * If the user passes the parameters explicitly, a new instance of the DBWorker is created upon each new request.
 84:  *
 85:  * Currently supported: MySQL und MS SQL.
 86:  *
 87:  * @param array $parameters
 88:  * Connection settings as an associative array in the form key => value:
 89:  *
 90:  * - $parameters["db_type"] - type of the database (MySQL or MSSQL)
 91:  *
 92:  * - $parameters["db_server"] - server address
 93:  *
 94:  * - $parameters["db_name"] - database name
 95:  *
 96:  * - $parameters["db_user"] - user name
 97:  *
 98:  * - $parameters["db_password"] - user password
 99:  *
100:  * - $parameters["autoconnect"] - should true if the connection should be established automatically
101:  *                 upon creation.
102:  *
103:  * Example:
104:  * ```php
105:  * $dbw = dbworker(["db_type" => "MySQL",
106:  *                  "db_server" => "localhost",
107:  *                  "db_name" => "framework_demo", 
108:  *                  "db_user" => "root", 
109:  *                  "db_password" => "root",
110:  *                  "autoconnect" => true
111:  *                 ]);
112:  * ```
113:  *
114:  * @return DatabaseWorkers\DBWorker|null
115:  * returns DBWorker object or null if the object could not be created.
116:  *
117:  * @author Oleg Schildt
118:  */
119: function dbworker($parameters = null)
120: {
121:   $msgmanager = singleton(IMessageManager::class);
122:   $lngmanager = singleton(ILanguageManager::class);
123: 
124:   $from_settings = false;
125:   
126:   if(empty($parameters))
127:   {
128:     $parameters["db_type"] = config_settings()->getParameter("db_type");
129:     $parameters["db_server"] = config_settings()->getParameter("db_server");
130:     $parameters["db_name"] = config_settings()->getParameter("db_name");
131:     $parameters["db_user"] = config_settings()->getParameter("db_user");
132:     $parameters["db_password"] = config_settings()->getParameter("db_password");
133:     $parameters["autoconnect"] = true;
134:     
135:     $from_settings = true;
136:   }
137:   
138:   if(empty($parameters["db_type"]))
139:   {
140:     $msgmanager->setError($lngmanager->text("ErrDatabaseTypeEmpty"));
141:     return null;
142:   }
143:   
144:   $class_name = "SmartFactory\\DatabaseWorkers\\" . $parameters["db_type"] . "_DBWorker";
145:   
146:   if($from_settings)
147:   {
148:     // if the dbworker is requested from the conenction settings, it should be a singleton.
149:     $dbworker = FactoryBuilder::getInstance($class_name, true);
150:   }
151:   else
152:   {
153:     // if the user passes the connection settings manually, create new instance each time.
154:     $dbworker = FactoryBuilder::getInstance($class_name, false);
155:   }
156:   
157:   if(!$dbworker->is_extension_installed())
158:   {
159:     $msgmanager->setError(sprintf($lngmanager->text("ErrDbExtenstionNotInstalled"), $dbworker->get_extension_name(), $dbworker->get_rdbms_name()));
160:     return null;
161:   }
162:   
163:   // do not connect, only object required
164:   // user will do connect by itself
165:   if(empty($parameters["autoconnect"]))
166:   {
167:     $dbworker->clear_messages();
168:     return $dbworker;
169:   }
170: 
171:   // instance already connected
172:   if($dbworker->is_connected())
173:   {
174:     $dbworker->clear_messages();
175:     return $dbworker;
176:   }
177:   
178:   // try to connect only if first time
179:   // if сonnection alredy tried and failed
180:   // do not try again within one request
181:   if($dbworker->get_last_error_id() != "") return false;
182: 
183:   $dbworker->init($parameters);
184: 
185:   if($dbworker->connect())
186:   {
187:     return $dbworker;
188:   }
189: 
190:   if($dbworker->get_last_error_id() == "conn_data_err")
191:   {
192:     $msgmanager->setError($lngmanager->text("ErrNoDBConnectionData"));
193:   }
194:   elseif($dbworker->get_last_error_id() == "conn_err")
195:   {
196:     $msgmanager->setError($lngmanager->text("ErrDbInaccessible"),
197:                              sprintf($lngmanager->text("ErrDbConnNoAccess"), checkempty($parameters["db_server"]), checkempty($parameters["db_user"]))
198:                             );
199:   }
200:   elseif($dbworker->get_last_error_id() == "db_err")
201:   {
202:     $msgmanager->setError($lngmanager->text("ErrDbInaccessible"),
203:                              sprintf($lngmanager->text("ErrDbConnNoDB"), checkempty($parameters["db_name"]), checkempty($parameters["db_user"]))
204:                             );
205:   }
206:   
207:   return false;
208: } // instance
